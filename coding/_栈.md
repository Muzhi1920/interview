
- 移除K个数字使得值更小
```cpp
for (char c : num) {
    //从前往后，有比当前大的就替换。保证排在前面的更小
    while (k && res.size() && res.back() > c)
        res.pop_back();
        --k;
    if (res.size()==0 && c== '0')
        continue;
    res.push_back(c);
}
while (res.size() && k--) 
    res.pop_back();
return res.empty() ? "0" : res;
```
